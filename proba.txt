    <script>
        // Vanta.js pozadina
        window.addEventListener('load', () => {
            VANTA.WAVES({
                el: "#vanta-bg",
                mouseControls: true,
                touchControls: true,
                gyroControls: false,
                minHeight: 200.00,
                minWidth: 200.00,
                scale: 1.00,
                scaleMobile: 1.00,
                color: 0x222650
            });
        });

        // Countdown timer
        const countdownDate = new Date("Sep 30, 2025 00:00:00").getTime();
        const countdownFunction = setInterval(() => {
            const now = new Date().getTime();
            const distance = countdownDate - now;

            const days = Math.floor(distance / (1000 * 60 * 60 * 24));
            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((distance % (1000 * 60)) / 1000);

            document.getElementById("countdown-days").innerHTML = String(days).padStart(2, '0');
            document.getElementById("countdown-hours").innerHTML = String(hours).padStart(2, '0');
            document.getElementById("countdown-minutes").innerHTML = String(minutes).padStart(2, '0');
            document.getElementById("countdown-seconds").innerHTML = String(seconds).padStart(2, '0');

            if (distance < 0) {
                clearInterval(countdownFunction);
                document.getElementById("countdown-days").innerHTML = "00";
                document.getElementById("countdown-hours").innerHTML = "00";
                document.getElementById("countdown-minutes").innerHTML = "00";
                document.getElementById("countdown-seconds").innerHTML = "00";
            }
        }, 1000);

        // Text slider
        const texts = ["DIGITAL STRATEGY", "CREATIVE DEVELOPMENT", "BRANDING & MARKETING"];
        let textIndex = 0;
        const textSlider = document.getElementById("text-slider").querySelector("p");

        const updateText = () => {
            textSlider.style.opacity = '0';
            setTimeout(() => {
                textSlider.textContent = texts[textIndex];
                textSlider.style.opacity = '1';
                textIndex = (textIndex + 1) % texts.length;
            }, 1000);
        };

        updateText();
        setInterval(updateText, 4000);
    </script>
